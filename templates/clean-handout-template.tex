% Just a simple box for highlighting things; has a blueish border
% Takes an optional argument that provides a title
\newmdenv[style=rcbox,%
outerlinewidth = 1,%
backgroundcolor = gray!10,%
outerlinecolor = blue!70!black,%
linecolor = blue!70!black,%
]{hlbox}
%
\makeatletter
\let\orig@hlbox=\hlbox
\def\hlbox{
  \@ifnextchar[{\hlbox@opt}{\orig@hlbox}
}
\def\hlbox@opt[#1]{
  \orig@hlbox[frametitle={#1}]
}
\makeatother
%
